## Task 1: Mastery Dashboard — add `?activity=dashboard` to exercises.js

**Files:** `js/exercises.js` (extend)

**Action:**
Add `runDashboard(app)` function. Read SRS state from `SRS.summary()`. Display:
- Three stat blocks: New / Learning / Mastered word counts
- A visual "bank of successes" — a row of coloured dots, one per mastered word (green filled) vs not yet mastered (grey)
- A plain-English progress note — no punitive framing
- Buttons: "Start flashcards", "Take a quiz", "Back to exercises"

```javascript
function runDashboard(app) {
  fetch('data/vocabulary/all.json')
    .then(r => r.json())
    .then(words => {
      const summary = SRS.summary();
      const total = words.length;
      const mastered = summary.mastered;
      const learning = summary.learning;
      const newCount = total - mastered - learning;

      // Build dot strip (max 50 dots shown for layout)
      const displayCount = Math.min(total, 100);
      const dots = [];
      for (let i = 0; i < displayCount; i++) {
        if (i < mastered) dots.push('<span class="dash-dot mastered" title="Mastered"></span>');
        else if (i < mastered + learning) dots.push('<span class="dash-dot learning" title="Learning"></span>');
        else dots.push('<span class="dash-dot new" title="Not yet seen"></span>');
      }
      if (total > displayCount) dots.push(`<span class="dash-dot-more">+${total - displayCount} more</span>`);

      const progressNote = mastered === 0
        ? 'Start with flashcards — every word you rate correctly moves up a level.'
        : mastered >= total
          ? 'Outstanding — every word mastered. Keep reviewing to stay sharp.'
          : `You\'ve secured ${mastered} out of ${total} words. Keep going!`;

      app.innerHTML = `
        <div class="ex-header">
          <a href="quiz.html" class="ex-back">← Exercises</a>
          <span class="ex-ce-label">Your progress</span>
        </div>
        <h2>Mastery Dashboard</h2>
        <div class="dash-stats">
          <div class="dash-stat dash-stat--mastered">
            <span class="dash-stat-num">${mastered}</span>
            <span class="dash-stat-label">Mastered</span>
          </div>
          <div class="dash-stat dash-stat--learning">
            <span class="dash-stat-num">${learning}</span>
            <span class="dash-stat-label">Learning</span>
          </div>
          <div class="dash-stat dash-stat--new">
            <span class="dash-stat-num">${newCount}</span>
            <span class="dash-stat-label">Not yet seen</span>
          </div>
        </div>
        <div class="dash-dots">${dots.join('')}</div>
        <p class="dash-note">${progressNote}</p>
        <div class="dash-actions">
          <a href="quiz.html?activity=flashcard" class="ex-btn ex-btn-primary">Flashcards</a>
          <a href="quiz.html?activity=mcq" class="ex-btn ex-btn-secondary">MCQ Quiz</a>
          <a href="quiz.html" class="ex-btn ex-btn-secondary">All exercises</a>
        </div>
      `;
    })
    .catch(() => {
      app.innerHTML = '<p class="error">Could not load progress data.</p>';
    });
}
```

```javascript
// At the top of runFlashcard() and runMCQ(), after loading vocabulary:
const params = new URLSearchParams(location.search);
const topic = params.get('topic');
const pos   = params.get('pos');
const decl  = params.get('decl') ? parseInt(params.get('decl')) : null;
const conj  = params.get('conj') ? parseInt(params.get('conj')) : null;
const freq  = params.get('freq') ? parseInt(params.get('freq')) : null;

let pool = words;
if (topic)  pool = pool.filter(w => w.topics && w.topics.includes(topic));
if (pos)    pool = pool.filter(w => w.part_of_speech === pos);
if (decl)   pool = pool.filter(w => w.declension === decl);
if (conj)   pool = pool.filter(w => w.conjugation === conj);
if (freq)   pool = pool.filter(w => w.frequency_rank && w.frequency_rank <= freq);

if (pool.length < 4) pool = words; // fallback if filter too narrow
```

After each grammar group is rendered, call a helper:

```javascript
function appendQuickCheck(container, checkKey) {
  const check = GRAMMAR_QUICK_CHECKS[checkKey];
  if (!check) return;
  let answered = false;

  const div = document.createElement('div');
  div.className = 'qcheck';
  const opts = [...check.options]; // already in good order, shuffle for fairness:
  const shuffled = [...opts].sort(() => Math.random() - 0.5);
  div.innerHTML = `
    <p class="qcheck-label">Quick check</p>
    <p class="qcheck-q">${check.question}</p>
    <div class="qcheck-opts">
      ${shuffled.map(o => `<button class="qcheck-btn" data-val="${o}">${o}</button>`).join('')}
    </div>
    <div class="qcheck-fb" style="display:none"></div>
  `;
  container.appendChild(div);

  div.querySelectorAll('.qcheck-btn').forEach(btn => {
    btn.addEventListener('click', () => {
      if (answered) return;
      answered = true;
      const correct = btn.dataset.val === check.correct;
      div.querySelectorAll('.qcheck-btn').forEach(b => {
        b.disabled = true;
        if (b.dataset.val === check.correct) b.classList.add('qcheck-correct');
        else if (b === btn && !correct) b.classList.add('qcheck-wrong');
      });
      const fb = div.querySelector('.qcheck-fb');
      fb.style.display = 'block';
      fb.innerHTML = `<p class="${correct ? 'fb-correct' : 'fb-wrong'}">${correct ? '✓ Correct!' : `✗ The answer is <strong>${check.correct}</strong>`}</p><p class="fb-explanation">${check.explanation}</p>`;
    });
  });
}
```
